# PRINCIPE :
# définir la zone de dessin et la valeur du zoom
# calculer la taille de l'image en fonction de ces deux paramètres

# AVANTAGES :
# L"utilisateur choisi lui-même l'échelle (le zoom)
# La fractale conserve toujours les mêmes proportions

# INCONVÉNIENTS :
# La taille de l'image n'est pas connue à l'avance sans la calculer
# Risques d'avoir soit une image trop grande, soit trop petite
# Temps de dessin conséquent pour l'image trop grande

# PALLIER AUX DÉFAUTS :
# Demander une confirmation de dessin en affichant la taille de l'image (avec une fenêtre d'info par exemple)



// définition de la zone de dessin, ici la fractale en entier
définir xmin 	      = -2.1
définir xmax 	      = 0.6
définir ymin 	      = -1.2
définir ymax 	      = 1.2
définir zoom 	      = 100 	// pour une distance de 1 sur le plan, on a 100 pixel sur l'image
définir iteration_max = 50

// calcul de la taille de l'image
définir image_x = (xmax - xmin) * zoom
définir image_y = (ymax - ymin) * zoom

Pour x = 0 tant que x < image_x par pas de 1 
    Pour y = 0 tant que y < image_y par pas de 1
        définir c_r = x / zoom + xmin	  // partie réelle
        définir c_i = y / zoom + ymin	  // partie imaginaire
        définir z_r = 0
        définir z_i = 0
        définir i   = 0

        Faire
            définir tmp = z_r
            z_r = z_r*z_r - z_i*z_i + c_r
            z_i = 2*z_i*tmp + c_i
            i   = i+1
        Tant que z_r*z_r + z_i*z_i < 4 et i < iteration_max

        si i = iteration_max
            dessiner le pixel de coordonné (x; y)
        finSi
    finPour
finPour
